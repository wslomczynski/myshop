CREATE TABLE SYSTEM(
NAME TEXT PRIMARY KEY NOT NULL,
DESCRIPTION TEXT NOT NULL,
TECHNOLOGY_DESCRIPTION TEXT NOT NULL,
OWNER TEXT NOT NULL)



CREATE TABLE AGREEMENT(
system_name TEXT references SYSTEM(NAME),
request BIGSERIAL PRIMARY KEY,
order_number TEXT NOT NULL,
from_date date NOT NULL,
to_date date NOT NULL,
amount real NOT NULL,
amount_type CHAR(10) NOT NULL,
amount_period CHAR(10) NOT NULL,
authorization_percent integer NOT NULL,
active boolean NOT NULL);




@ResponseBody
	@RequestMapping(value = "/model", method = RequestMethod.POST)
	public String model(@RequestParam("system_name") String system_name, @RequestParam("from_date") String from_date,
			@RequestParam("to_date") String to_date, @RequestParam("amount") double amount,
			@RequestParam("amount_type") String amount_type, @RequestParam("amount_period") String amount_period,
			@RequestParam("authorization_percent") int authorization_percent, @RequestParam("active") boolean active)
			throws ParseException {
		SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
		Date from = sdf.parse(from_date);
		Agreement agreement = new Agreement();
		agreement.setFrom_date(from);
		Date to = sdf.parse(to_date);
		agreement.setTo_date(to);
		agreement.setSystem_name(system_name);
		agreement.setAmount(amount);
		agreement.setAmount_type(amount_type);
		agreement.setAmount_period(amount_period);
		agreement.setAuthorization_percent(authorization_percent);
		agreement.setActive(active);
		return "success";
